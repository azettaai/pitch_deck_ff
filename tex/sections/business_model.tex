\section{Business Model}

\subsection{Revenue Streams}

\begin{enumerate}
    \item \textbf{Model Usage Fees}: Consumption-based pricing for intelligence
    \begin{itemize}[noitemsep]
        \item Standard COEM Models: \$0.10 per 1M tokens
        \item Fine-Tuned Models: Custom pricing per 1M tokens
    \end{itemize}
    
    \item \textbf{Managed Infrastructure}: At-cost pass-through pricing
    \begin{itemize}[noitemsep]
        \item \textit{Cloud Cost}: Customer pays raw cloud provider rate (e.g., AWS/GCP instance cost) with no markup.
        \item \textit{Billing}: Annual commitment preferred.
        \item \textit{Flexibility}: Zero-fee instance scaling (upgrade/downgrade) enabled by decoupled storage architecture.
    \end{itemize}
    
    \item \textbf{Professional Services}: Integration, training, custom models
\end{enumerate}

\subsection{Unit Economics by Segment}

\subsubsection{Individuals \& Hobbyists}

\textbf{Revenue Model:}
\begin{itemize}[noitemsep]
    \item Free Tier: Shared CPU + 10,000 embeddings/month (unlimited searches, \$0 revenue)
    \item Starter: \$9/month - Shared CPU + 50,000 embeddings (unlimited searches)
    \item Pro: \$29/month - 1 vCPU + 200,000 embeddings (unlimited searches)
    \item Overage: \$0.10 per 10K embeddings beyond quota
\end{itemize}

\textbf{Cost Analysis (per paying user):}
\begin{itemize}[noitemsep]
    \item Infrastructure (shared compute): \$2-4/month
    \item Model inference (GPU, batched): \$1-3/month
    \item Storage \& bandwidth: \$0.50-2/month
    \item \textbf{Total COGS:} \$3.50-9/month
    \item \textbf{Gross margin:} 0-70\% (Free tier subsidized by paid users)
\end{itemize}

\textbf{Conversion assumptions:} 5-10\% of free users convert to paid tiers.

\subsubsection{Small Startups}

\textbf{Revenue Model:}
\begin{itemize}[noitemsep]
    \item Growth: \$99/month - 1 vCPU instance + 500K embeddings (unlimited searches)
    \item Business: \$299/month - 2 vCPU instance + 2M embeddings (unlimited searches)
    \item Overage: \$0.10 per 10K embeddings beyond quota
\end{itemize}

\textbf{Cost Analysis (per customer):}
\begin{itemize}[noitemsep]
    \item Infrastructure (1-2 vCPU): \$20-40/month
    \item Model inference (GPU): \$10-25/month
    \item Storage \& bandwidth: \$5-10/month
    \item Support overhead: \$5-10/month (email support)
    \item \textbf{Total COGS:} \$40-85/month
    \item \textbf{Gross margin:} 60-75\%
\end{itemize}

\textbf{Revenue projections:}
\begin{itemize}[noitemsep]
    \item Average deal size: \$150-250/month (including overage)
    \item CAC: \$500-1,000 (self-service, marketing-driven)
    \item Payback period: 4-6 months
    \item Annual LTV: \$1,800-3,000 (assuming 12-18 month retention)
\end{itemize}

\subsection{Key Assumptions}

\begin{enumerate}[noitemsep]
    \item Customer acquisition ramps progressively: 200 → 800 → 2,500 → 5,000 → 8,000 total customers
    \item Average revenue per customer grows from \$520/month (Y1) to \$1,275/month (Y5)
    \item Gross margin improves from 55\% (Y1) to 70\% (Y5) through scale efficiencies
    \item Churn rate: 5\%/month (hobbyists), 3\%/month (SMB), 1\%/month (enterprise), 0.5\%/month (tech giants)
    \item Infrastructure costs decrease 15-20\% annually through optimization and better pricing
    \item Sales efficiency improves 30\% year-over-year as product matures
\end{enumerate}
