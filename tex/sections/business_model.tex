\section{Business Model}

\subsection{Revenue Streams}

\begin{enumerate}
    \item \textbf{Model Usage Fees}: Consumption-based pricing for intelligence
    \begin{itemize}[noitemsep]
        \item Standard COEM Models: \$0.10 per 1M tokens
        \item Fine-Tuned Models: Custom pricing per 1M tokens
    \end{itemize}
    
    \item \textbf{Managed Infrastructure}: At-cost pass-through pricing
    \begin{itemize}[noitemsep]
        \item \textit{Cloud Cost}: Customer pays raw cloud provider rate (e.g., AWS/GCP instance cost) with no markup.
        \item \textit{Billing}: Annual commitment preferred.
        \item \textit{Flexibility}: Zero-fee instance scaling (upgrade/downgrade) enabled by decoupled storage architecture.
    \end{itemize}
    
    \item \textbf{Professional Services}: Integration, training, custom models
\end{enumerate}
