\subsection{Search Features}

\begin{itemize}[noitemsep]
    \item \textbf{Automatic Embedding}: Text automatically converted to vectors
    \item \textbf{Metadata Filtering}: Structured filters (date, category, author)
    \item \textbf{Semantic Re-ranking}: Distance + coherence + language priors
    \item \textbf{Result Normalization}: Consistent scoring across languages
\end{itemize}
