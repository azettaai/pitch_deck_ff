\section{System Architecture}

\subsection{High-Level Overview}

Figure \ref{fig:system-overview} illustrates Cosma's four-layer architecture designed for enterprise deployment.

\begin{figure}[ht]
\centering
\begin{tikzpicture}[
    node distance=1.5cm, 
    scale=0.95, 
    every node/.style={font=\small},
    % Enhanced component styles with richer gradients
    client/.style={
        rectangle, rounded corners=4pt, 
        minimum width=2.0cm, minimum height=0.8cm, align=center, 
        draw=cosmaPurple!85, line width=0.8pt,
        top color=cosmaPurple!2, bottom color=cosmaPurple!25,
        drop shadow={shadow xshift=0.12cm, shadow yshift=-0.12cm, opacity=0.25}
    },
    process/.style={
        rectangle, rounded corners=4pt, 
        minimum width=2.5cm, minimum height=1.0cm, align=center, 
        draw=cosmaGray!85, line width=0.8pt,
        top color=cosmaGray!2, bottom color=cosmaGray!25,
        drop shadow={shadow xshift=0.12cm, shadow yshift=-0.12cm, opacity=0.25}
    },
    component/.style={
        rectangle, rounded corners=4pt, 
        minimum width=3.0cm, minimum height=1.0cm, align=center, 
        draw=cosmaBlue!85, line width=0.8pt,
        top color=cosmaBlue!3, bottom color=cosmaBlue!28,
        drop shadow={shadow xshift=0.15cm, shadow yshift=-0.15cm, opacity=0.3}
    },
    api/.style={
        rectangle, rounded corners=4pt, 
        minimum width=2.5cm, minimum height=1.0cm, align=center, 
        draw=cosmaOrange!85, line width=0.8pt,
        top color=cosmaOrange!3, bottom color=cosmaOrange!28,
        drop shadow={shadow xshift=0.12cm, shadow yshift=-0.12cm, opacity=0.25}
    },
    storage/.style={
        cylinder, shape border rotate=90, aspect=0.25, 
        minimum width=2.5cm, minimum height=1.2cm, align=center, 
        draw=cosmaGreen!85, line width=0.8pt,
        top color=cosmaGreen!3, bottom color=cosmaGreen!28,
        drop shadow={shadow xshift=0.12cm, shadow yshift=-0.12cm, opacity=0.25}
    },
    layer/.style={
        draw=gray!50, dashed, rounded corners=6pt, 
        top color=gray!3, bottom color=gray!12, 
        inner sep=0.4cm,
        drop shadow={shadow xshift=0.1cm, shadow yshift=-0.1cm, opacity=0.15}
    },
    layerlabel/.style={
        font=\bfseries\small, 
        text=white,
        rectangle, rounded corners=3pt,
        inner sep=3pt,
        top color=gray!70, bottom color=gray!90
    },
    arrow/.style={thick, ->, >=stealth, color=black!60, line width=1.2pt}
]

% Client Layer
\node (web) [client] {Web UI};
\node (mobile) [client, right=0.3cm of web] {Mobile SDK};
\node (api) [client, right=0.3cm of mobile] {REST API};
\node (cli) [client, right=0.3cm of api] {CLI};

\begin{scope}[on background layer]
    \node (clientlayer) [layer, fit=(web) (mobile) (api) (cli), label={[layerlabel]above:Client Layer}] {};
\end{scope}

% Application Layer
\node (search) [component, below=2.0cm of mobile] {Search Service};
\node (billing) [component, left=0.5cm of search] {Billing};
\node (monitor) [component, right=0.5cm of search] {Monitoring};

\begin{scope}[on background layer]
    \node (applayer) [layer, fit=(billing) (search) (monitor), label={[layerlabel]above:Application Layer}] {};
\end{scope}

% Model Layer
\node (embedapi) [api, below=2.0cm of billing] {Embedding API};
\node (coem) [component, below=0.5cm of embedapi] {COEM Models};
\node (inference) [process, right=0.5cm of coem] {GPU Inference};

\begin{scope}[on background layer]
    \node (modellayer) [layer, fit=(embedapi) (coem) (inference), label={[layerlabel]above:Model Layer}] {};
\end{scope}

% Storage Layer
\node (cvd) [component, below=2.0cm of coem] {CVD Engine};
\node (postgres) [storage, below=0.5cm of cvd] {PostgreSQL};
\node (index) [process, right=0.5cm of postgres] {Vector Index};

\begin{scope}[on background layer]
    \node (storagelayer) [layer, fit=(cvd) (postgres) (index), label={[layerlabel]above:Storage Layer}] {};
\end{scope}

% Enhanced connections between layers with gradient effect
\draw [arrow] (clientlayer.south) -- (applayer.north);
\draw [arrow] (applayer.south) -- (modellayer.north);
\draw [arrow] (applayer.south) -- (storagelayer.north);
\draw [arrow] (modellayer.south) -- (storagelayer.north);

\end{tikzpicture}
\caption{Cosma System Architecture - Four-Layer Design}
\label{fig:system-overview}
\end{figure}
